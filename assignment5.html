def ExtractArticles(urlBase, pages = 1):
    """
    Input: The url of the article list.
           The number of pages to fetch links from (default = 1).
    Output: A list of article urls.
    """
    urlList = []
    
    # Grab urls from each page:
    for page in xrange(1, pages + 1):
        
        # Correct url if not page 1:
        if page > 1:
            url = urlBase + "page/" + str(page) + "/"
        else:
            url = urlBase
            
        # Get html and parse:
        reqArticle = requests.get(url)
        artsHTML = reqArticle.text
        arts = BeautifulSoup(artsHTML)
        
        # Grab the urls we want:
        main = arts.find(name = 'main')
        the15 = main.find_all(name = "div", class_ = "post-thumb preload image-loading zoom-img-in")
        for chunk in the15: urlList.append(chunk.a['href'])
    
    return urlList

def ArticleInfo(url):
    """
    Input: URL of article.
    Output: A dictionary with keys "url", "title", "text", and "author" and the respective values. 
    """
    # Get html and parse:
    reqArticle = requests.get(url)
    artHTML = reqArticle.text
    artParse = BeautifulSoup(artHTML)
    
    # Get title:
    title = artParse.find(name = "h1", class_ = "entry-title", itemprop = "headline").text
    
    # Get text and author:
    allText= u''
    artBody = artParse.find(name = "div", itemprop = "articleBody")
    # For each paragraph in the body:
    paras = artBody.find_all(name = "p")
    for para in paras:
        text = para.text
        # If author line put in author, otherwise put in text:
        found = text.lower().find("written by")
        if found != -1:
            allText = allText + text[:found]
            author = text[found:]
        elif paras[-1] == para: # If didn't find author through written by, take last line:
            author = text
        else:
            allText = allText + text
    
    # Put in dictionary:
    artDict = {'url': url, 'title': title, 'author': author, 'text': allText.translate({ 0x2018:0x27, 0x2019:0x27, 0x201C:0x22, 0x201D:0x22, 0x2026:0x20 })
    }
    
    return artDict

def GetNews(url, numArticles, cat):
    """
    Input: URL for where to get the news articles. 
           The number of articles to get. 
           The category of news (a string). 
    Output: A data frame with five columns: category, author, text, title, and url. 
    """
    DF = pd.DataFrame()
    urlList = ExtractArticles(url, int(np.ceil(numArticles / 15.0)) )  
    for eachURL in urlList:
        info = ArticleInfo(eachURL)
        info["category"] = cat
        asDF = pd.DataFrame.from_dict([info])
        DF = DF.append(asDF)
    return DF

def NLP(colName, newColName, news):  
    """
    Input: colName = name of column to take words from.
           newColName = what to call the new column of words.
           news = the news dataframe.
    Output: news = dataframe with the extra column of words. 
    """
    # Words that don't tell us anything about the topic of the article:
    stopWords = ["a", "about","also","an", "and","any", "are", "as","at","be", "been", "before", "but", "by", "do","for", "from", "has", "have", "i", "if", "in", "is","it", "of", "or", "on", "one","our", "so","some","that", "the","then", "this", "to", "too", "was", "way", "we","what","will", "why", "with", u'\u2014', ".", ",", ":", "$", "---", "--" "'s", "''", "``", "'", "[", "]", "(", ")"]
    newDF = pd.DataFrame(columns = [newColName])
    for index, row in news.iterrows():
        tokens = tokenize(row[colName]) 
        tokens = [t.lower() for t in tokens if t.lower() not in stopWords]
        tempDict = {newColName: tokens}
        tempDF = pd.DataFrame.from_dict([tempDict])
        newDF = newDF.append(tempDF)
        
    newDF = newDF.reset_index(drop=True)
    return pd.concat([news, newDF], axis=1)

def plotWords(countTuple, titleAddOn):
    """
    Input: countTuples = list of tuples with word and count.
           
    Output: A horizontal bar plot of the 15 most common words from the titles of the articles.
    """
    temp = pd.DataFrame.from_records(countTuple, columns = ["word", "count"])
    ax = sns.factorplot(x = "count", y = "word", data = temp, kind = "bar", orient = "h")
    ax.set(xlabel = "Count", ylabel = "Topic")
    plt.title("15 Most Common Topics From " + titleAddOn)
