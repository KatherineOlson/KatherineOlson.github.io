def ndb_search(searchTerm, key):
    """
    Input: The search term and API key
    Output: The search result items as a list (for 0 items, return an empty list).
    """
    base =  "https://api.nal.usda.gov/ndb/search/"
    base = base + "?" + "format=json"
    base = base + "&q=" + searchTerm
    base = base + "&api_key=" + key
    response = requests.get(base)
    response = response.json()
    return response

def getNDBNums(df, searchCol, key):
    """
    Input: df = a starting data frame.
           searchCol = the name of the column in the df with what to search for.
           key = the API key.
    Output: the same df but with a new column ndb.
    """
    types = df["type"]
    cols = ("food", "ndbNum")
    foodNDB = pd.DataFrame(columns = cols)
    for fruitVeg in fresh[searchCol]: #49
        # Get search results:
        resultJSON = ndb_search(fruitVeg, key)
        
        # Elimate search results with words that we don't want:
        dfOptions = pd.DataFrame(resultJSON["list"]["item"])
        badWords = ["juice", "sparkling", "with", "and ", "&", "sauce", "soda", 
                    "syrup", "bread", "drink", " seed", " gum", " sour", "flavor", 
                   "yogurt", "beverage", "punch", "cooler", "tea", "snack", 
                   "pureed", "kool-aid", "fruit roll"]
        for word in badWords:
            dfOptions = dfOptions[dfOptions["name"].str.lower().str.find(word) == -1]

        # If it contains the word raw, then only keep those:
        if any(dfOptions["name"].str.lower().str.find("raw")) != -1:
            dfOptions = dfOptions[dfOptions["name"].str.lower().str.find("raw") != -1]

        # If any left are in the Fruits and Fruit Juices group, pick those:    
        if any(dfOptions["group"] == "Fruits and Fruit Juices"):
            dfOptions = dfOptions[dfOptions["group"] == "Fruits and Fruit Juices"] 
        # Same for veggies:
        elif any(dfOptions["group"] == "Vegetables and Vegetable Products"):
            dfOptions = dfOptions[dfOptions["group"] == "Vegetables and Vegetable Products"] 
            
        # If it has the food name then keep those (sweet corn has diff order so need if):
        if any(dfOptions["name"].str.lower().str.find(fruitVeg) != -1) :
            dfOptions = dfOptions[dfOptions["name"].str.lower().str.find(fruitVeg) != -1]

        # Now grab the first one -> most relevant search result once the changes
        # above are made. Seems to be the best one to pick for most cases.
        if (dfOptions.shape[0] > 0):
            ndbNum = dfOptions.iloc[0]["ndbno"]
        else:
            ndbNum = np.nan # problem: equal to previous
            
        # Now add to dataframe with just food and nbdNum
        newFood = pd.DataFrame([[fruitVeg, ndbNum]], columns = cols)
        foodNDB = foodNDB.append(newFood)
        
    # Now merege with original data frame
    dfMerge = pd.merge(df, foodNDB)
    
    return dfMerge

def nutrition(fresh):
    """
    Input: The fresh data frame.
    Output: The same data frame with four more columns: energy, nutritionEst, 
    nutritionEstG, and nutritionEstB (for good and bad). 
    """
    cols = ["food", "energy", "nutritionEst", "nutritionEstG", "nutritionEstB"]
    newColsDF = pd.DataFrame(columns = cols)
    good = ["Protein", "Fiber, total dietary", "Fatty acids, total monounsaturated", 
            "Fatty acids, total polyunsaturated", "Calcium, Ca", "Iron, Fe", 
             "Magnesium, Mg", "Phosphorus, P", "Potassium, K", "Zinc, Zn", 
             "Thiamin", "Riboflavin", "Vitamin B-6", 
            "Vitamin C, total ascorbic acid", "Vitamin E (alpha-tocopherol)"]
    bad = ["Total lipid (fat)", "Carbohydrate, by difference", "Sugars, total", 
           "Fatty acids, total saturated", "Fatty acids, total trans", 
            "Sodium, Na", "Cholesterol"]
    # Go through each ndpNum and get energy and nutritionEst
    for num, food in zip(fresh.ix[:, "ndbNum"], fresh["food"]):
        
        # Only want ndpNums that are not NaN:
        if np.isnan(float(num)) == False: 
            
            # Get API info and read into data frame:
            jsonInfo = ndb_report(fresh[fresh["food"] == food]["ndbNum"].iloc[0])
            info = pd.DataFrame(jsonInfo)
            
            # Get energy:
            energy = float(info[info["name"] == "Energy"]["value"]) # .iloc[0])
            
            # Get nutrition estimate:
            nutEstG, nutEstB, nutEst = 0, 0, 0
            goodInfo = [info[info["name"] == g] for g in good]
            badInfo = [info[info["name"] == b] for b in bad]
            for gi in goodInfo + badInfo:
                # If that food has a value for that nutrition type:
                if gi["value"].shape[0] > 0 : 
                    value = float(gi["value"].iloc[0])
                    # Convert to grams if necessary:
                    if gi["unit"].iloc[0] == "mg":
                        value = value / 1000
                    # If good nutrition type, then add:
                    if any([gi["name"].iloc[0] == g for g in good]):
                        nutEstG = nutEstG + value
                        nutEst = nutEst + value
                    else: # Bad then subtract:
                        nutEstB = nutEstB + value
                        nutEst = nutEst - value
                # Not found so zero grams:             
                
            tempdf = pd.DataFrame([[food, energy, nutEst, nutEstG, nutEstB]], columns = cols)
            newColsDF = newColsDF.append(tempdf)
    
    return pd.merge(newColsDF, fresh)
